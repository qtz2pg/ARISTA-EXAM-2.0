! Host and management details
hostname border1
ip routing

! Loopback interfaces
interface Loopback0
  ip address 10.200.200.1/32

! Routed links to core/external
interface Ethernet1
  no switchport
  ip address 10.200.1.1/30
  ip ospf network point-to-point
  no shutdown

interface Ethernet2
  no switchport
  ip address 10.200.2.1/30
  ip ospf network point-to-point
  no shutdown

! OSPF configuration
router ospf 1
  router-id 10.200.200.1
  network 10.200.200.1/32 area 0.0.0.0
  network 10.200.1.1/30 area 0
  network 10.200.2.1/30 area 0
  passive-interface default
  no passive-interface Ethernet1
  no passive-interface Ethernet2

! BGP configuration
router bgp 65500
  router-id 10.200.200.1
  maximum-paths 2 ecmp 2
  neighbor CORE peer group
  neighbor CORE remote-as 65100
  neighbor CORE update-source Loopback0
  neighbor 10.200.1.2 peer group CORE
  neighbor 10.200.1.2 description Core1
  neighbor 10.200.2.2 peer group CORE
  neighbor 10.200.2.2 description Core2
  network 10.200.200.1/32
  redistribute connected

! VLANs
vlan 10
  name HOST1

! Spanning tree configuration
spanning-tree mode mstp
spanning-tree mst 0 vlan 1-4094

! Optional: Management, NTP, SNMP, security, etc.

end
