hostname Spine2

! Interface Configuration
! -----------------------------

   no switchport
   ip address 10.10.2.1/30
   no shutdown

interface Ethernet2
   no switchport
   ip address 10.10.2.5/30
   no shutdown

interface Ethernet3
   no switchport
   ip address 10.10.2.9/30
   no shutdown

   no switchport
   ip address 10.10.2.13/30
   no shutdown
